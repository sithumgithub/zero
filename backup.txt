process

ps - process in current shell
ps -A  - all active processes
ps -au  - ll process and process not associated with terminal
ps -ef  - full format
ps -x  - all the process owned by the current user
ps -fu username/grpname/root  - display users, process by real user id or name.
ps -e --forest  - process tree
ps -eo pid,ppid,cmd,%mem,%cpu --sort=-%mem | head  - top running process
kill -o pid  - stop process
